  FUN signed int testBreakWhile 
    #0: signed int a
  BEGIN
    BREAK
    WHILE (*a)
      *a = foo(*a)
    END_WHILE
    BREAK    
  END

  FUN signed int testBreakDoWhile 
    #0: signed int a
  BEGIN
    BREAK
    DO
      *a = foo(*a)
    WHILE (*a)
    BREAK    
  END

  FUN signed int testBreakFor 
    #0: signed int a
  BEGIN
    BREAK
    signed int r = \
      0
    signed int i
    FOR (*i = 0; *i < *a; *i += 1)
      *r += foo(*i)
    END_FOR
    BREAK    
  END

  FUN signed int testBreakSwitch 
    #0: signed int a
    #1: signed int b
  BEGIN
    BREAK
    signed int r = \
      0
    SWITCH (*a)
      CASE 1: *r = foo(*b * *b)
      BREAK
      CASE 2: *r = foo(*b * *b)
      BREAK
      CASE 3: *r = foo(*b * *b)
      BREAK
      CASE 4: *r = foo(*b * *b)
      BREAK      
    END_SWITCH
    BREAK
    RETURN *r
  END
