    *STRUCT S1 s1
    *STRUCT S2 s2
    *STRUCT S3 s3

      *s1 - *s2
    signed int b = \
      *s3 - *s3_2
    signed int c = \
      *s1 - *s3
    *signed int ip
    *double dp
    **signed int ipp
    signed int d = \
      *ip - *dp
    signed int e = \
      *ip - *ipp
    signed int[1] arr
    (*signed int)[1] parr
    signed int f = \
      *dp - arr
    signed int g = \
      arr - parr
    *void vp
    signed int h = \
      *vp - *ip
  END
