  FUN void forTest 
    #0: signed int a
    #1: signed int b
  BEGIN
    signed int i = \
      0
    FOR (; i < a; ++i)
      IF (a > b)
      THEN
        foo(a - b)
        CONTINUE        
      END_IF
      foo(b)
      foo(a)
      forTest(b, b - a)

  END

  FUN void whileTest 
    #0: signed int a
    #1: signed int b
  BEGIN
    signed int i = \
      0

      IF (a > b)
      THEN
        foo(a - b)
        ++i
        CONTINUE        
      END_IF
      foo(b)
      foo(a)
      whileTest(b, b - a)

  END
